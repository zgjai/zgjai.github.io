func Join

func Join(a []string, sep string) string
返回值：以sep为分隔符将a拼接起来。

实现原理：

// 先计算出拼接完成后string的长度
n := len(sep) * (len(a) - 1)
for i := 0; i < len(a); i++ {
    n += len(a[i])
}

b := make([]byte, n)
bp := copy(b, a[0])
for _, s := range a[1:] {
    bp += copy(b[bp:], sep)
    bp += copy(b[bp:], s)
}

func Map

func Map(mapping func(rune) rune, s string) string
返回值：对s中的每个字符进行mapping，如果mapping结果为负则丢弃，构成新的string。

实现原理：

func Map(mapping func(rune) rune, s string) string {
    // In the worst case, the string can grow when mapped, making
    // things unpleasant. But it's so rare we barge in assuming it's
    // fine. It could also shrink but that falls out naturally.
    maxbytes := len(s) // length of b
    nbytes := 0        // number of bytes encoded in b
    // The output buffer b is initialized on demand, the first
    // time a character differs.
    var b []byte

    for i, c := range s {
        r := mapping(c)
        // 这个if是应对mapping之后跟原字符串完全相同的情况
        if b == nil {
            if r == c {
                continue
            }
            b = make([]byte, maxbytes)
            nbytes = copy(b, s[:i])
        }
        if r >= 0 {
            wid := 1
            // 超出了ASCII范围
            if r >= utf8.RuneSelf {
                wid = utf8.RuneLen(r)
            }
            if nbytes+wid > maxbytes {
                // Grow the buffer.
                maxbytes = maxbytes*2 + utf8.UTFMax
                nb := make([]byte, maxbytes)
                copy(nb, b[0:nbytes])
                b = nb
            }
            nbytes += utf8.EncodeRune(b[nbytes:maxbytes], r)
        }
    }
    if b == nil {
        return s
    }
    return string(b[0:nbytes])
}

func Repeat

func Repeat(s string, count int) string
返回值：s重复count次组成的string。

实现原理：

func Repeat(s string, count int) string {
    // Since we cannot return an error on overflow,
    // we should panic if the repeat will generate
    // an overflow.
    // See Issue golang.org/issue/16237
    if count < 0 {
        panic("strings: negative Repeat count")
    } else if count > 0 && len(s)*count/count != len(s) {
       // 需要注意检查长度是不是会溢出
        panic("strings: Repeat count causes overflow")
    }

    // 直接创建一个最终长度的byte slice
    // 不仅仅是为了避免扩容，也是为了求得正确结果的必要条件
    b := make([]byte, len(s)*count)
    bp := copy(b, s)
    for bp < len(b) {
       // 指数级增长，如果count不是2的整数幂，最后一次copy时会被截断到正确的长度
        copy(b[bp:], b[:bp])
        bp *= 2
    }
    return string(b)
}

func Replace

func Replace(s, old, new string, n int) string
返回值：这个还是需要动手实验一下才行，各种情况还是比较多的。

实现原理：

// Replace returns a copy of the string s with the first n
// non-overlapping instances of old replaced by new.
// If old is empty, it matches at the beginning of the string
// and after each UTF-8 sequence, yielding up to k+1 replacements
// for a k-rune string.
// If n < 0, there is no limit on the number of replacements.
func Replace(s, old, new string, n int) string {
    if old == new || n == 0 {
        return s // avoid allocation
    }

    // Compute number of replacements.
    // Count可以计算出s中old出现了多少次
    if m := Count(s, old); m == 0 {
        return s // avoid allocation
    } else if n < 0 || m < n {
       // n为实际替换次数
        n = m
    }

    // Apply replacements to buffer.
    t := make([]byte, len(s)+n*(len(new)-len(old)))
    w := 0
    start := 0
    for i := 0; i < n; i++ {
        j := start
        if len(old) == 0 {
            // old 为空时，在每两个字符中间添加new
            if i > 0 {
                // 计算字符长度
                _, wid := utf8.DecodeRuneInString(s[start:])
                j += wid
            }
        } else {
            // 定位到下一个old
            j += Index(s[start:], old)
        }
        // copy不需要被替换掉部分
        w += copy(t[w:], s[start:j])
        // 添加new
        w += copy(t[w:], new)
        start = j + len(old)
    }
    w += copy(t[w:], s[start:])
    return string(t[0:w])
}

几个Split中共用的部分：

// s是源字符串，sep是分隔字符串，sepSave是保留sep的多少位，n是分隔后的slice的最长长度
func genSplit(s, sep string, sepSave, n int) []string {
        if n == 0 {
            return nil
        }
        if sep == "" {
            return explode(s, n)
        }
        if n < 0 {
           // n<0表示完全分隔，计算出分隔后的字符串的数量
            n = Count(s, sep) + 1
        }
        c := sep[0]
        start := 0
        a := make([]string, n)
        na := 0
        for i := 0; i+len(sep) <= len(s) && na+1 < n; i++ {
            if s[i] == c && (len(sep) == 1 || s[i:i+len(sep)] == sep) {
                a[na] = s[start : i+sepSave]
                na++
                // start是每个分隔出来的字符串的起始位置
                start = i + len(sep)
                // i是每个分隔出来的字符串的结束位置
                i += len(sep) - 1
            }
        }
        // 剩余部分
        a[na] = s[start:]
        return a[0 : na+1]
}

展开字符串函数：

func explode(s string, n int) []string {
    // 获取s中unicode字符数量
        l := utf8.RuneCountInString(s)
        if n < 0 || n > l {
            n = l
        }
        a := make([]string, n)
        for i := 0; i < n-1; i++ {
           // 拆解
            ch, size := utf8.DecodeRuneInString(s)
            a[i] = s[:size]
            s = s[size:]
            if ch == utf8.RuneError {
                a[i] = string(utf8.RuneError)
            }
        }
        if n > 0 {
            a[n-1] = s
        }
        return a
}

func Title

func Title(s string) string
返回值："her royal highness" -> "Her Royal Highness"

实现原理：

// Title returns a copy of the string s with all Unicode letters that begin words
// mapped to their title case.
//
// BUG(rsc): The rule Title uses for word boundaries does not handle Unicode punctuation properly.
func Title(s string) string {
    // Use a closure here to remember state.
    // Hackish but effective. Depends on Map scanning in order and calling
    // the closure once per rune.
    prev := ' '
    return Map(
        func(r rune) rune {
          // 闭包体现在变量prev上
          // 个人觉得闭包的本质，就是将某些原本不属于某个函数的变量绑定到了这个函数上
          // 比如这里的prev跟这个匿名函数的关系
            if isSeparator(prev) {
                prev = r
                return unicode.ToTitle(r)
            }
            prev = r
            return r
        },
        s)
}

func TrimLeftFunc

func TrimLeftFunc(s string, f func(rune) bool) string
返回值：将s从开头部分开始，移除符合f的字符，直至遇到第一个不符合f的即停止。

实现原理：

// TrimLeftFunc returns a slice of the string s with all leading
// Unicode code points c satisfying f(c) removed.
func TrimLeftFunc(s string, f func(rune) bool) string {
    // 找到第一个不符合f的字符index
    i := indexFunc(s, f, false)
    if i == -1 {
        return ""
    }
    // 切割
    return s[i:]
}

func TrimRightFunc

func TrimRightFunc(s string, f func(rune) bool) string
返回值：将s从结尾部分开始，移除符合f的字符，直至遇到第一个不符合f的即停止。

实现原理：

// TrimRightFunc returns a slice of the string s with all trailing
// Unicode code points c satisfying f(c) removed.
func TrimRightFunc(s string, f func(rune) bool) string {
    // 从尾部找到第一个不符合f的字符index
    i := lastIndexFunc(s, f, false)
    if i >= 0 && s[i] >= utf8.RuneSelf {
        _, wid := utf8.DecodeRuneInString(s[i:])
        i += wid
    } else {
        i++
    }
    // 切割，需要注意的是前闭后开，故有上面那一段
    return s[0:i]
}

func TrimLeft

func TrimLeft(s string, cutset string) string
返回值：将s从开头部分开始，移除存在于cutset中的字符，直至遇到第一个不存在的即停止。

实现原理：

// TrimLeft returns a slice of the string s with all leading
// Unicode code points contained in cutset removed.
func TrimLeft(s string, cutset string) string {
    if s == "" || cutset == "" {
        return s
    }
    return TrimLeftFunc(s, makeCutsetFunc(cutset))
}

func makeCutsetFunc(cutset string) func(rune) bool {
    if len(cutset) == 1 && cutset[0] < utf8.RuneSelf {
       // cutset只有1个字符且为ASCII
        return func(r rune) bool {
            return r == rune(cutset[0])
        }
    }
    if as, isASCII := makeASCIISet(cutset); isASCII {
       // 全为ASCII时，使用makeASCIISet，之前用到过
        return func(r rune) bool {
            return r < utf8.RuneSelf && as.contains(byte(r))
        }
    }
    return func(r rune) bool { return IndexRune(cutset, r) >= 0 }
}

func TrimRight

func TrimRight(s string, cutset string) string
返回值：将s从结尾部分开始，移除存在于cutset中的字符，直至遇到第一个不存在的即停止。

实现原理：

// TrimRight returns a slice of the string s, with all trailing
// Unicode code points contained in cutset removed.
func TrimRight(s string, cutset string) string {
    if s == "" || cutset == "" {
        return s

func TrimPrefix(s, prefix string) string {
    // 判断是否包含前缀
